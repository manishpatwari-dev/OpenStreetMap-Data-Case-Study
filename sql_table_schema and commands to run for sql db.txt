CREATE TABLE nodes (
    changeset INTEGER,
	id INTEGER PRIMARY KEY NOT NULL,
    lat REAL,
    lon REAL,
    timestamp TEXT,
	uid INTEGER,
	user TEXT,
    version INTEGER
);

CREATE TABLE nodes_tags (
    id INTEGER,
    key TEXT,
	type TEXT,
    value TEXT,
    FOREIGN KEY (id) REFERENCES nodes(id)
);

CREATE TABLE ways (
    changeset INTEGER,
    id INTEGER PRIMARY KEY NOT NULL,
    timestamp TEXT,
    uid INTEGER,
    user TEXT,
    version TEXT
);

CREATE TABLE ways_tags (
    id INTEGER NOT NULL,
    key TEXT NOT NULL,
    type TEXT,
    value TEXT NOT NULL,
    FOREIGN KEY (id) REFERENCES ways(id)
);

CREATE TABLE ways_nodes (
    id INTEGER NOT NULL,
    node_id INTEGER NOT NULL,
    position INTEGER NOT NULL,
    FOREIGN KEY (id) REFERENCES ways(id),
    FOREIGN KEY (node_id) REFERENCES nodes(id)
);
